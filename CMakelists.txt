# -----------------------------------------------------------------------------
# CMake & Project Configuration
# -----------------------------------------------------------------------------
cmake_minimum_required(VERSION 3.16)
project(MyCoolProject VERSION 1.0.0 LANGUAGES CXX C)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)

if(MSVC)
  set(CMAKE_MSVC_RUNTIME_LIBRARY "MultiThreaded$<$<CONFIG:Debug>:Debug>")
endif()
set(BUILD_SHARED_LIBS OFF)

# -----------------------------------------------------------------------------
# Dependency Management (using FetchContent)
# -----------------------------------------------------------------------------
include(FetchContent)

# --- glm ---
FetchContent_Declare(
  glm
  GIT_REPOSITORY https://github.com/g-truc/glm.git
  GIT_TAG        1.0.1
)

# --- glfw ---
set(GLFW_BUILD_EXAMPLES OFF)
set(GLFW_BUILD_TESTS OFF)
set(GLFW_BUILD_DOCS OFF)
FetchContent_Declare(
  glfw
  GIT_REPOSITORY https://github.com/glfw/glfw.git
  GIT_TAG        3.4
)

FetchContent_MakeAvailable(glm glfw)

# --- Assimp ---
set(ASSIMP_BUILD_TESTS OFF)
set(ASSIMP_BUILD_ASSIMP_TOOLS OFF)
FetchContent_Declare(
  assimp
  GIT_REPOSITORY https://github.com/assimp/assimp.git
  GIT_TAG        v6.0.2 # Or another official release tag
)

FetchContent_MakeAvailable(glm glfw assimp)

# -----------------------------------------------------------------------------
# Build GLEW from local source
# -----------------------------------------------------------------------------
add_library(glew_static STATIC "${CMAKE_CURRENT_SOURCE_DIR}/thirdParty/glew/src/glew.c")

target_include_directories(glew_static PUBLIC
  "${CMAKE_CURRENT_SOURCE_DIR}/thirdParty/glew/include"
)

target_compile_definitions(glew_static PUBLIC GLEW_STATIC)

if(WIN32)
  target_link_libraries(glew_static PUBLIC opengl32)
endif()

# -----------------------------------------------------------------------------
# ## NEW: Build Dear ImGui from local source
# -----------------------------------------------------------------------------
# Assumes ImGui is located in "thirdParty/imgui"
add_library(imgui STATIC
    "${CMAKE_CURRENT_SOURCE_DIR}/thirdParty/imgui/imgui.cpp"
    "${CMAKE_CURRENT_SOURCE_DIR}/thirdParty/imgui/imgui_draw.cpp"
    "${CMAKE_CURRENT_SOURCE_DIR}/thirdParty/imgui/imgui_tables.cpp"
    "${CMAKE_CURRENT_SOURCE_DIR}/thirdParty/imgui/imgui_widgets.cpp"
    "${CMAKE_CURRENT_SOURCE_DIR}/thirdParty/imgui/imgui_demo.cpp" 
    "${CMAKE_CURRENT_SOURCE_DIR}/thirdParty/imgui/backends/imgui_impl_glfw.cpp"
    "${CMAKE_CURRENT_SOURCE_DIR}/thirdParty/imgui/backends/imgui_impl_opengl3.cpp"
)

target_link_libraries(imgui PUBLIC glfw)

# Make ImGui's headers available to any target that links it
target_include_directories(imgui PUBLIC
    "${CMAKE_CURRENT_SOURCE_DIR}/thirdParty/imgui"
    "${CMAKE_CURRENT_SOURCE_DIR}/thirdParty/imgui/backends"
)

# -----------------------------------------------------------------------------
# Executable Configuration
# -----------------------------------------------------------------------------

add_executable(PhysicsTest "")

# This will hide the console ONLY in Release mode on Windows
if(MSVC)
    target_link_options(PhysicsTest PRIVATE
        "$<$<CONFIG:Release>:/SUBSYSTEM:WINDOWS>"
        "$<$<CONFIG:Release>:/ENTRY:mainCRTStartup>"
    )
endif()

# Find all .cpp files in 'src' and add them to the executable
file(GLOB_RECURSE SOURCES CONFIGURE_DEPENDS "src/*.cpp")
target_sources(PhysicsTest PRIVATE ${SOURCES})

# Tell the project where to find its own header files (e.g., for #include "MyHeader.h")
target_include_directories(PhysicsTest PRIVATE
  "${CMAKE_CURRENT_SOURCE_DIR}/include"
)

# -----------------------------------------------------------------------------
# Linking Libraries
# -----------------------------------------------------------------------------
# ## MODIFIED: Added 'imgui' to the list of linked libraries
target_link_libraries(PhysicsTest PRIVATE
  glew_static
  glm
  glfw
  imgui
  assimp
)

# -----------------------------------------------------------------------------
# Post-Build Commands
# -----------------------------------------------------------------------------
add_custom_command(TARGET PhysicsTest POST_BUILD
    COMMAND ${CMAKE_COMMAND} -E copy_directory
    "${CMAKE_SOURCE_DIR}/src/Shaders" # Source
    "$<TARGET_FILE_DIR:PhysicsTest>/Shaders" # Destination
    COMMENT "Copying shaders to build output directory..."
)

add_custom_command(TARGET PhysicsTest POST_BUILD
    COMMAND ${CMAKE_COMMAND} -E copy_directory
    "${CMAKE_SOURCE_DIR}/res" # Source
    "$<TARGET_FILE_DIR:PhysicsTest>/res" # Destination
    COMMENT "Copying resources to build output directory..."
)

file(COPY res DESTINATION ${CMAKE_BINARY_DIR})
